   {
       for(int j=num;j>i;j--)
       {
           printf(" ");
       }
       for(int k=1;k<=2*i-1;k++)
       {
           printf("*");
       }
      printf("\n");
   }

输出三行四列的数组并求取最大值
    int b[3][4],max=0,x=0,y=0;
    printf("请输入一个3行4列的数组\n");
    for (int i=0; i<3; i++) {
        for (int j=0; j<4; j++) {
            scanf("%d",&b[i][j]);
            printf("%d  ",b[i][j]);
            if (b[i][j]>max) {
                max=b[i][j];
                x=i+1;
                y=j+1;
            }
        }
        printf("\n");
    }
    printf("该数组中的最大值为%d\n它的行号是%d\n它的列号是%d\n",max,x,y);
