//@protected 受保护的，OC默认属性修饰，只能被本类和子类访问
//@public       公共的，所有的地方都可以使用
//@privated    私有的，只可以在本类使用
//@package    一般用于系统库的封装中
//@property    
@interface Time:NSObject
{
     @property int _hour;
     int _minute;
     int _second;
}
//设置时间的方法(setter)
-(void)setHour:(int)hour;
-(void)setHour:(int)hour;
-(void)setHour:(int)hour;
//显示时间
-(void)showTime;
@end
@implementation Time
-(void)showTime
{
     NSLog(@“now is %d:%d:%d”,_hour,_minute,_second);
}
@end
