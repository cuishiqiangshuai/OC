// 不可变字符串 NSString
        //1.实例方法创建
        NSString *str1=@"hello";
        NSString *str2=[[NSString alloc] initWithCString:"qingyun" encoding:NSUTF8StringEncoding];//encoding指的是编码方式
        //alloc init==new  new只是有一种初始化方式，但是alloc init可以有多种初始化方式 initWithXXXX
        NSString *str3=[[NSString alloc] initWithUTF8String:"china"];
        //str2和str3的方式都一样：将C的字符串编码为OC的字符串
        NSString *str4=[[NSString alloc] initWithString:str1];
        NSLog(@"str1:%@,str2:%@,str3:%@,str4:%@",str1,str2,str3,str4);
        NSLog(@"str1:%p,str2:%p,str3:%p,str4:%p",str1,str2,str3,str4);
        //改变指针的指向
        str2=@"ios";
        NSLog(@"str2:%@,%p",str2,str2);
        //从文件获取
        NSError *error;
        NSString *str5=[[NSString alloc] initWithContentsOfFile:@"/Users/qingyun/Desktop/张小东/Code/0401/OC-Foundation(字符串)/str1.txt" encoding:NSUTF8StringEncoding error:&error];
        if (!error)
        {
            NSLog(@"str5:%@,%p",str5,str5);
        }
        else
        {
            NSLog(@"error:%@",error);
        }
        //类方法
        NSString *str6=[NSString stringWithFormat:@"i %@ %s very much",@"like","qingyun"];
        NSLog(@"str6==>%@",str6);//Format表示按照你想要的格式去创建对象
       
        //字符串对象的操作
        //1.长度
        NSLog(@"长度>>%ld,%ld",[str6 length],str5.length);
        //2.比较
        if ([str6 isEqualToString:str5])
        {
            NSLog(@"str6==str5");
        }
        else
        {
            NSLog(@"str6 != str5");
        }
        NSComparisonResult result=[str5 compare:str6];//比较的是ASCII
        NSLog(@"result==%ld",result);//-1表示前者小，0表示相等，1表示前者大
        switch (result)
        {
            case NSOrderedAscending://升序
                NSLog(@"str5 < str6");
                break;
            case NSOrderedSame:
                NSLog(@"same");
                break;
            case NSOrderedDescending://降序
                NSLog(@"str5 > str6");
                break;
               
            default:
                break;
        }
        NSString *myStr=@"Hello,Qingyun";
        //NSLiteralSearch区分大小写(默认)，NSCaseInsensitiveSearch不区分大小写
        result=[str5 compare:myStr options:(NSLiteralSearch)];
        NSLog(@"result=>%ld",result);
        switch (result)
        {
            case NSOrderedAscending:
                NSLog(@"str5 < myStr");
                break;
            case NSOrderedSame:
                NSLog(@"same");
                break;
            case NSOrderedDescending:
                NSLog(@"str5 > myStr");
                break;
               
            default:
                break;
        }
        result=[@"hello32a" compare:@"hello23" options:NSNumericSearch];
        NSLog(@"result==>%ld",result);
        //包含
        if ([str5 containsString:@"hello"])
        {
            NSLog(@"contain");
        }
        //查找
        NSRange range1=[str5 rangeOfString:@"qingyun"];//9223372036854775807表示NSNotFound没找到
        NSLog(@"range1==>%@",NSStringFromRange(range1));
        //拼接
        NSString *newStr=[str5 stringByAppendingString:@" ios"];
        NSLog(@"%@",newStr);
        //截取
        NSString *newStr1=[str5 substringFromIndex:1];//从某个位置开始截取
        NSLog(@"newStr1=>%@",newStr1);
        newStr1=[newStr1 substringToIndex:4];//到哪结束//注意不要越界！
        NSLog(@"newStr1==>%@",newStr1);
        //等同于下面一句代码
        NSLog(@"newStr2==>%@",[str5 substringWithRange:NSMakeRange(1, 4)]);
        //路径
        NSString *pathStr1=@"~/Desktop/Code/20160401/OC-Foundation(字符串)/str1.txt";
        NSLog(@"++>%@",[pathStr1 stringByExpandingTildeInPath]);//路径展开，绝对路径
        //后缀
        NSLog(@">>%@",[pathStr1 pathExtension]);
        //转化为相对路径
        NSLog(@">>>%@",[@"/Users/qingyun/Desktop/Code/20160401/OC-Foundation(字符串)/str1.txt" stringByAbbreviatingWithTildeInPath]);
